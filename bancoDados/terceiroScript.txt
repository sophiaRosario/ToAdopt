create database toadopt;
use toadopt; 

create table usuario(
  idUsuario int primary key auto_increment,
  nome VARCHAR(45),
  sobrenome VARCHAR(45),
  email VARCHAR(45),
  telefone CHAR(11),
  senha CHAR(8)
  );
-- alter table usuario rename column telCel to telefone;

create table adocao(
  idAdocao int primary key auto_increment,
  nomeAnimal VARCHAR(45),
  especie CHAR(8),
  sexo CHAR(5),
  porte CHAR(7),
  fkUsuario int,
  foreign key (fkUsuario)
    references usuario(idUsuario)
);

create table respostaQuiz(
  idQuiz int primary key auto_increment,
  totalPontos int,
  fkUsuario int,
  foreign key (fkUsuario)
    references usuario(idUsuario)
);

create table comentario (
  idComentario int primary key,
  titulo varchar(100),
  descricao varchar(200),
  fkUsuario int,
  foreign key (fkUsuario)
    references usuario(idUsuario)
   );
